SET(LIB_NAME LZMA)

SET(C_FILES
	${SEVENZIP_PROJECT_DIR}/C/Alloc.c
	${SEVENZIP_PROJECT_DIR}/C/CpuArch.c
	${SEVENZIP_PROJECT_DIR}/C/LzFind.c
	${SEVENZIP_PROJECT_DIR}/C/LzmaDec.c
	${SEVENZIP_PROJECT_DIR}/C/LzmaEnc.c
	${SEVENZIP_PROJECT_DIR}/C/LzmaLib.c
)
IF(WIN32)
	SET(C_FILES ${C_FILES}
		${SEVENZIP_PROJECT_DIR}/C/Util/LzmaLib/LzmaLibExports.c
	)
ENDIF()

SET(DEF_FILES
	${SEVENZIP_PROJECT_DIR}/C/Util/LzmaLib/LzmaLib.def
)
SET(LZMA_HEADER_FILES
	${SEVENZIP_PROJECT_DIR}/C/7zTypes.h
	${SEVENZIP_PROJECT_DIR}/C/Alloc.h
	${SEVENZIP_PROJECT_DIR}/C/CpuArch.h
	${SEVENZIP_PROJECT_DIR}/C/LzFind.h
	${SEVENZIP_PROJECT_DIR}/C/LzHash.h
	${SEVENZIP_PROJECT_DIR}/C/LzmaDec.h
	${SEVENZIP_PROJECT_DIR}/C/LzmaEnc.h
	${SEVENZIP_PROJECT_DIR}/C/LzmaLib.h
)
SOURCE_GROUP("Source Files" FILES ${DEF_FILES} ${C_FILES})
SOURCE_GROUP("Header Files" FILES ${LZMA_HEADER_FILES})

ADD_LIBRARY(${LIB_NAME} ${ZENGINE_PREFERRED_LIB_TYPE}
	${C_FILES} ${DEF_FILES} ${LZMA_HEADER_FILES}
)

target_compile_definitions(${LIB_NAME}
	PRIVATE
		-DLZMALIB_EXPORTS
		-DZ7_ST
)
target_include_directories(${LIB_NAME}
	PUBLIC
		${SEVENZIP_PROJECT_DIR}
)

SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES
	OUTPUT_NAME ${LIB_NAME}${ZENGINE_OUTPUT_SUFFIX}
	FOLDER "3rd/7z"
)

TARGET_LINK_LIBRARIES(${LIB_NAME})
